const work = function(n1,n2,...args) {
    
    let sum1 = n1 + n2; 
    let sum2 = 1;
    for(let arg of args){
        sum2 *= arg;
    }
    let final = [sum1,sum2];
    return final;
}    
console.log(work(3,4,6,7,8));
